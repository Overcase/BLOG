// 配置pug为模版引擎

const Koa = require('koa')

const app = new Koa()

const views = require('koa-views')

const { resolve } = require("path")

app.use(views(resolve(__dirname, './views'), {

extension: 'pug' //后缀名称

}))

app.use(async (ctx, next) => {

  await ctx.render('index', {
  
  you: 'luck',
  
  me: 'Scott'
  
  })
})

app.listen(3000)
